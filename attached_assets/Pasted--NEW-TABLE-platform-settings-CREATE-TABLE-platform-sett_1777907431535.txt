## NEW TABLE: platform_settings

CREATE TABLE platform_settings (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  setting_key text UNIQUE NOT NULL,
  setting_value text,
  setting_type text CHECK (setting_type IN (
    'text', 'email', 'url', 'number',
    'boolean', 'color', 'json', 'file_url'
  )) DEFAULT 'text',
  category text CHECK (category IN (
    'branding',
    'email',
    'notifications',
    'commerce',
    'legal',
    'features'
  )) NOT NULL,
  label text NOT NULL,
  description text,
  is_public boolean DEFAULT false,
  last_edited_by uuid REFERENCES team(id),
  updated_at timestamptz DEFAULT now()
);

ALTER TABLE platform_settings
  ENABLE ROW LEVEL SECURITY;

CREATE POLICY "super_admin_settings_all"
ON platform_settings FOR ALL TO authenticated
USING (get_my_role() = 'super_admin')
WITH CHECK (get_my_role() = 'super_admin');

CREATE POLICY "authenticated_read_public_settings"
ON platform_settings FOR SELECT TO authenticated
USING (is_public = true);

## SETTINGS UTILITY

Create: src/lib/settings.ts

import { supabase } from './supabase'

type SettingsCache = Record<string, string>
let cache: SettingsCache | null = null
let cacheLoadedAt: number | null = null
const CACHE_TTL_MS = 5 * 60 * 1000

export async function getSettings(
  keys?: string[]
): Promise<SettingsCache> {
  const now = Date.now()
  if (
    cache &&
    cacheLoadedAt &&
    now - cacheLoadedAt < CACHE_TTL_MS
  ) {
    if (!keys) return cache
    return Object.fromEntries(
      keys.map(k => [k, cache![k] ?? ''])
    )
  }

  const query = supabase
    .from('platform_settings')
    .select('setting_key, setting_value')

  const { data } = await query
  cache = Object.fromEntries(
    (data || []).map(s => [s.setting_key, s.setting_value ?? ''])
  )
  cacheLoadedAt = now

  if (keys) {
    return Object.fromEntries(
      keys.map(k => [k, cache![k] ?? ''])
    )
  }
  return cache
}

export async function getSetting(key: string)
  : Promise<string> {
  const settings = await getSettings([key])
  return settings[key] ?? ''
}

export function invalidateSettingsCache() {
  cache = null
  cacheLoadedAt = null
}

Use these functions everywhere settings are needed:

const platformName = await getSetting('platform_name')
const currency = await getSetting('default_currency')
const fobMin = await getSetting('fob_minimum_price_usd')

## SESSION TIMEOUT

Implement idle session timeout based on
session_timeout_minutes setting:

In a top-level React component (App.tsx or Layout):

let idleTimer: ReturnType<typeof setTimeout>

async function resetIdleTimer() {
  clearTimeout(idleTimer)
  const minutes = parseInt(
    await getSetting('session_timeout_minutes')
  ) || 120
  idleTimer = setTimeout(async () => {
    await supabase.auth.signOut()
    router.replace('/login?reason=timeout')
  }, minutes * 60 * 1000)
}

const events = ['mousedown', 'keydown',
  'touchstart', 'scroll']
events.forEach(e =>
  document.addEventListener(e, resetIdleTimer))

## SETTINGS VALIDATION

Add client-side validation before saving:

primary_color / accent_color:
  Must match /^#[0-9A-Fa-f]{6}$/

email fields:
  Must match standard email regex

url fields:
  Must start with https:// or be empty

fob_minimum_price_usd:
  Must be >= 100 and <= 10000

session_timeout_minutes:
  Must be >= 30 and <= 480

accepted_currencies:
  Must be comma-separated 3-letter codes
  Validate each: /^[A-Z]{3}$/

admin_notification_emails:
  Must be comma-separated valid email addresses
  Validate each entry

Show inline validation errors in red below
each field as the user types.
Disable save button until all validations pass.
