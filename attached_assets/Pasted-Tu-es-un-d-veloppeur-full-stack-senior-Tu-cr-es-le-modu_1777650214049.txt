model Client {

  id                String   @id @default(uuid())
  companyName       String   // NOM DU TIERS *
  altName           String?  // NOM ALTERNATIF
  type              String   // Entreprise, Particulier, Association, Administration
  clientCode        String   @unique // CODE CLIENT auto: CLI-YYYY-XXXX
  
  // === ADRESSE ===
  address           String?
  postalCode        String?
  city              String?
  country           String   // PAYS *
  region            String?  // Département / État / Province
  
  // === CONTACT ===
  phone             String?
  mobile            String?
  fax               String?
  website           String?
  email             String?
  refuseMassEmail   Boolean  @default(false)
  
  // === IDENTIFIANTS FISCAUX (adaptés par pays) ===
  proId1            String?  // SIRET, EIN, Cégjegyzékszám, etc.
  proId2            String?  // SIREN, Adószám, etc.
  vatRegistered     Boolean  @default(false)
  vatNumber         String?  // TVA intracommunautaire, GST, etc.
  
  // === ORIGINE ===
  source            String   // kompass, salon, referral, web, import_excel, converted_prospect
  convertedFromId   String?  // ID du prospect d'origine (si conversion)
  
  // === CATÉGORISATION ===
  tags              String[] // Vanille, Importateur, Distributeur, B2B, etc.
  internalNotes     String?  // Notes internes commerciales
  
  // === DONNÉES COMMERCIALES ===
  activityType      String?  // importateur, distributeur, transformateur, industriel, artisan, autre
  riskLevel         String   @default("medium") // low, medium, high, blocked
  creditLimit       Float?   // Limite de crédit en devise principale
  paymentTerms      Int      @default(30) // Jours de paiement
  preferredCurrency String   @default("USD") // USD, EUR, GBP
  preferredIncoterm String?  // FOB, CIF, EXW, DDP
  
  // === CONTACTS MULTIPLES (nouveau) ===
  contacts          Contact[] // Plusieurs contacts par client
  
  // === HISTORIQUE ===
  totalOrders       Int      @default(0)
  totalRevenue      Float    @default(0) // Cumulé toutes commandes
  lastOrderDate     DateTime?
  averageOrderValue Float    @default(0)
  
  // === STATUT ===
  isActive          Boolean  @default(true)
  createdBy         String   // userId
  assignedTo        String?  // Commercial responsable
  
  // === TIMESTAMPS ===
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt
  
  // === RELATIONS ===
  deals             Deal[]
  quotes            Quote[]
  orders            Order[]
  invoices          Invoice[]
  interactions      Interaction[]
}

model Contact {
  id          String   @id @default(uuid())
  clientId    String
  firstName   String
  lastName    String
  role        String?  // Directeur commercial, Acheteur, PDG, etc.
  email       String?
  phone       String?
  mobile      String?
  isPrimary   Boolean  @default(false) // Contact principal
  isActive    Boolean  @default(true)
  createdAt   DateTime @default(now())
  
  client      Client   @relation(fields: [clientId], references: [id])
}

2. CONVERSION PROSPECT → CLIENT (LOGIQUE MÉTIER)
================================================================

ENDPOINT : PUT /api/crm/prospects/:id/convert

CONDITIONS DE CONVERSION :
1. Prospect.status = "qualified" ET score >= 60
2. OU Prospect.status = "contacted" avec interaction positive
3. OU validation manuelle par Super Admin (override)

PROCESSUS DE CONVERSION :

Étape 1 — Vérification
  - Prospect existe et non déjà converti
  - Prospect.convertedToClientId = null
  
Étape 2 — Création Client
  - Copier tous les champs identiques (companyName, address, etc.)
  - Générer clientCode : CLI-YYYY-NNNN (auto-incrément annuel)
  - Source = "converted_prospect"
  - convertedFromId = prospect.id
  
Étape 3 — Migration données liées
  - Déplacer interactions prospect → client
  - Déplacer deals prospect → client
  - Déplacer quotes prospect → client
  - Créer contact principal depuis prospect.contact
  
Étape 4 — Mise à jour Prospect
  - status = "converted"
  - convertedToClientId = client.id
  
Étape 5 — Actions automatiques
  - Créer Deal "Première commande" si aucun deal existant
  - Créer Reminder "Onboarding client" (7 jours)
  - Log interaction "Conversion prospect → client"
  - Notification au commercial assigné

RÉPONSE API :
{
  success: true,
  client: { ... },
  message: "Prospect converti en client CLI-2026-0042",
  nextSteps: ["Créer un devis", "Planifier appel onboarding"]
}

================================================================
3. CONVERSION AUTOMATIQUE (COMMANDE/DEVIS VALIDÉ)
================================================================

ENDPOINT : POST /api/crm/orders (création commande)

LOGIQUE AUTO-CONVERSION :

Si commande créée avec prospectId (pas clientId) :
  1. Vérifier prospect.status = "qualified" ou "contacted"
  2. Exécuter conversion silencieuse
  3. Remplacer prospectId par clientId dans la commande
  4. Continuer création commande normalement

Si devis accepté (quote.status = "accepted") :
  1. Vérifier prospect lié au devis
  2. Convertir prospect → client
  3. Créer commande liée au nouveau client

7. RÈGLES MÉTIER & SÉCURITÉ
================================================================

CRÉATION CLIENT :
- Code client unique et auto-généré
- Email unique par client (si renseigné)
- Vérification TVA si pays UE et vatRegistered=true

MODIFICATION CLIENT :
- Super Admin : tout modifiable
- Commercial assigné : contacts, notes, tags modifiables
- Comptable : données fiscales modifiables
- Log de toutes modifications (audit trail)

RISQUE CLIENT :
- low : paiement dans les 15j, pas de litige
- medium : paiement 30j standard
- high : retard paiement > 60j, litige en cours
- blocked : impayé > 90j, aucune commande possible

CONVERSION :
- Prospect doit avoir score >= 60 OU validation admin
- Historique prospect conservé (archivé, pas supprimé)
- Interactions migrées vers client
