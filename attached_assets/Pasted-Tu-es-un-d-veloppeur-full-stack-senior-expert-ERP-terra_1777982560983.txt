model OperationReport {

-----------------------------------

model OperationLotStatus {
  id            String   @id @default(uuid())
  reportId      String
  lotId         String

  status        String   // processing, phenole, moldy, ready, preparing
  quantityKg    Float
}

-----------------------------------

model Consumable {
  id          String @id @default(uuid())
  name        String   // sachet_vide, papier_paraffine, carton, claie
  unit        String   // unité, kg, rouleau
  stock       Float
}

-----------------------------------

model ConsumableUsage {
  id            String @id @default(uuid())
  reportId      String
  consumableId  String
  quantityUsed  Float
}
