1. MODÈLES PRISMA
-----------------------------------

model Leave {
  id            String   @id @default(uuid())
  employeeId    String
  type          String   // annual, sick, unpaid
  startDate     DateTime
  endDate       DateTime
  days          Float

  status        String   // pending, approved, rejected

  reason        String?

  approvedBy    String?
  createdAt     DateTime @default(now())
}

-----------------------------------

model LeaveBalance {
  id           String @id @default(uuid())
  employeeId   String

  totalDays    Float
  usedDays     Float
  remaining    Float

  year         Int
}

-----------------------------------

model LeavePolicy {
  id           String @id @default(uuid())
  type         String
  daysPerYear  Float
  maxCarryOver Float
}
