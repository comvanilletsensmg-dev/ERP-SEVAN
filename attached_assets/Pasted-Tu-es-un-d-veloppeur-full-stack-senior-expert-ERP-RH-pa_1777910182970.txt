2. CALCUL PAIE (MADAGASCAR)
-----------------------------------

brut = salaireBase + primes

CNAPS = brut * 0.01
OSTIE = brut * 0.01

IRSA = calcul progressif

net = brut - CNAPS - OSTIE - IRSA
