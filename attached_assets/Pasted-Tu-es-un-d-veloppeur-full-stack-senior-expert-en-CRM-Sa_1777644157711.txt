model Prospect {
  id          String   @id @default(uuid())
  company     String
  contact     String?
  email       String?
  phone       String?
  country     String
  source      String   // kompass, salon, referral, web, manuel
  status      String   @default("new") // new, to_contact, contacted, qualified, converted, lost
  score       Int      @default(0)
  notes       String?
  assignedTo  String?  // userId du commercial
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model Client {
  id          String   @id @default(uuid())
  name        String
  country     String
  email       String?
  phone       String?
  currency    String   @default("USD") // USD, EUR, MGA
  riskLevel   String   @default("medium") // low, medium, high
  creditLimit Float?
  paymentTerms Int     @default(30) // jours
  isActive    Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model Deal {
  id           String   @id @default(uuid())
  title        String
  prospectId   String?
  clientId     String?
  stage        String   @default("prospect") // prospect, contact, negotiation, proposal, won, lost
  value        Float
  currency     String
  probability  Int      @default(20) // 0-100
  expectedClose DateTime?
  notes        String?
  assignedTo   String   // userId du commercial
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}

model Interaction {
  id          String   @id @default(uuid())
  type        String   // call, email, meeting, whatsapp, note
  note        String
  prospectId  String?
  clientId    String?
  dealId      String?
  createdBy   String   // userId
  createdAt   DateTime @default(now())
}

model Quote {
  id          String   @id @default(uuid())
  clientId    String
  dealId      String?
  totalHT     Float
  tva         Float
  totalTTC    Float
  currency    String
  status      String   @default("draft") // draft, sent, accepted, rejected, expired
  validUntil  DateTime?
  items       QuoteItem[]
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model QuoteItem {
  id          String   @id @default(uuid())
  quoteId     String
  lotId       String?
  description String
  quantity    Float
  unitPrice   Float
  total       Float
}

model Order {
  id          String   @id @default(uuid())
  clientId    String
  quoteId     String?
  dealId      String?
  totalHT     Float
  tva         Float
  totalTTC    Float
  currency    String
  status      String   @default("confirmed") // confirmed, prepared, shipped, delivered, cancelled
  incoterm    String   // FOB, CIF, EXW
  items       OrderItem[]
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model OrderItem {
  id          String   @id @default(uuid())
  orderId     String
  lotId       String
  quantity    Float
  unitPrice   Float
  total       Float
}

model Reminder {
  id          String   @id @default(uuid())
  type        String   // followup, payment, call, quote_expiry
  prospectId  String?
  clientId    String?
  dealId      String?
  note        String
  dueDate     DateTime
  status      String   @default("pending") // pending, done, cancelled
  createdBy   String
  createdAt   DateTime @default(now())
}

model EmailTemplate {
  id          String   @id @default(uuid())
  name        String   @unique
  subject     String
  body        String   // variables : {{company}}, {{contact}}, {{product}}, {{price}}
  category    String   // prospecting, followup, quote, payment
  createdAt   DateTime @default(now())
}

model EmailLog {
  id          String   @id @default(uuid())
  to          String
  subject     String
  templateId  String?
  status      String   // sent, failed, opened
  sentAt      DateTime @default(now())
}

3. LOGIQUE MÉTIER CRITIQUE
================================================================

A. CONVERSION PROSPECT → CLIENT :
   1. Prospect status = "qualified"
   2. Créer Client avec données du prospect
   3. Créer Deal lié au client
   4. Prospect status = "converted"
   5. Créer Reminder "premier contact client"

B. CRÉATION DEVIS :
   1. Vérifier stock disponible (lots status = "ready")
   2. Calculer TVA 20% (export = 0% si certificat)
   3. Générer numéro devis : DEV-2026-XXXX
   4. Créer écriture comptable DRAFT (non comptabilisée)

C. DEVIS → COMMANDE :
   1. Quote status = "accepted"
   2. Créer Order
   3. Réserver lots (status = "reserved")
   4. Générer facture (module comptabilité)
   5. Écriture comptable : Débit 411 / Crédit 701

D. SCORING AUTOMATIQUE PROSPECT :
   score = 0
   if country in ["USA","France","Germany","Netherlands"] → +30
   if source == "kompass" → +20
   if source == "referral" → +25
   if email && phone → +15
   if company contains "vanilla" or "spice" or "food" → +20
   score = min(score, 100)

E. RELANCES AUTOMATIQUES (CRON quotidien) :
   - Devis non répondus après 7 jours → reminder
   - Clients sans commande depuis 90 jours → reminder
   - Factures impayées après échéance → reminder + email

F. VALIDATIONS OBLIGATOIRES :
   - Vente impossible si stock insuffisant
   - Devis > 10 000 USD → validation Super Admin
   - Client risque "high" → alerte avant commande
