model Lot {
  id            String @id @default(uuid())

-----------------------------------

model ProductionTask {
  id            String @id @default(uuid())
  lotId         String
  type          String   // drying, sorting, packaging
  startDate     DateTime
  endDate       DateTime
  requiredStaff Int
}

-----------------------------------

model ExportOrder {
  id            String @id @default(uuid())
  clientId      String
  quantityKg    Float
  status        String   // pending, preparing, shipped
  deadline      DateTime
}

-----------------------------------

model TaskAssignment {
  id          String @id @default(uuid())
  taskId      String

}
